From virtual nodes to actual pixels: implementing flexbox-like layout in Zig

Introduction
#

In Part 1, we built the foundation: Node types, properties, and builder functions. Now we need to turn that abstract tree into actual pixels on screen.

The Renderer is responsible for:

  1. Layout: Calculating the position and size of each node
  2. Drawing: Rendering pixels to the screen
  3. Hit Testing: Determining which node the mouse is over
  4. Event Collection: Queuing click events for later processing

Let’s build it step by step, using our counter application as the running example.

// packages/jsx/src/renderer.zig

const std = @import("std");
const rl = @import("raylib");
const Node = @import("node.zig").Node;

pub const Renderer = struct {
    /// The screen area we're rendering into
    bounds: Rect,

    /// Array of event handler functions
    handlers: []const Handler,

    /// Buffer for converting strings to null-terminated (for raylib)
    text_buf: [512]u8 = undefined,

    /// Queue of handler indices to call after render
    events: [32]usize = undefined,
    event_count: usize = 0,

    /// Function pointer type for event handlers
    pub const Handler = *const fn () void;

    /// Initialize renderer with bounds and handler array
    pub fn init(bounds: Rect, handlers: []const Handler) Renderer {
        return .{
            .bounds = bounds,
            .handlers = handlers,
        };
    }

    /// Render the node tree to screen
    pub fn render(self: *Renderer, root: Node) void {
        // Clear event queue from previous frame
        self.event_count = 0;

        // Clear screen with background color
        rl.clearBackground(Theme.bg);

        // Recursively render the tree
        self.renderNode(root, self.bounds);
    }

    /// Call queued event handlers
    pub fn flush(self: *Renderer) void {
        for (0..self.event_count) |i| {
            const idx = self.events[i];
            if (idx < self.handlers.len) {
                self.handlers[idx]();
            }
        }
        self.event_count = 0;
    }
};

pub const Rect = struct {
    x: i32,
    y: i32,
    w: i32,
    h: i32,

    pub fn init(x: i32, y: i32, w: i32, h: i32) Rect {
        return .{ .x = x, .y = y, .w = w, .h = h };
    }

    /// Right edge (x + width)
    pub fn right(self: Rect) i32 {
        return self.x + self.w;
    }

    /// Bottom edge (y + height)
    pub fn bottom(self: Rect) i32 {
        return self.y + self.h;
    }

    /// Create inset rectangle (shrink by amount on all sides)
    pub fn inset(self: Rect, amount: i32) Rect {
        return .{
            .x = self.x + amount,
            .y = self.y + amount,
            .w = @max(0, self.w - 2 * amount),
            .h = @max(0, self.h - 2 * amount),
        };
    }

    /// Convert to raylib Rectangle (float-based)
    pub fn toRaylib(self: Rect) rl.Rectangle {
        return .{
            .x = @floatFromInt(self.x),
            .y = @floatFromInt(self.y),
            .width = @floatFromInt(self.w),
            .height = @floatFromInt(self.h),
        };
    }
};

fn renderNode(self: *Renderer, node: Node, bounds: Rect) void {
    switch (node.kind) {
        // Layout nodes
        .col => self.renderCol(node, bounds),
        .row => self.renderRow(node, bounds),
        .center => self.renderCenter(node, bounds),
        .stack => self.renderStack(node, bounds),
        .scroll => self.renderScroll(node, bounds),

        // Content nodes
        .text => self.renderText(node, bounds),
        .spacer => {}, // Spacers are invisible
        .divider => self.renderDivider(bounds),

        // Interactive nodes
        .button => self.renderButton(node, bounds),
        .toggle => self.renderToggle(node, bounds),
        .input => self.renderInput(node, bounds),
        .slider => {},

        // Container nodes
        .card => self.renderCard(node, bounds),
        .list => self.renderList(node, bounds),
        .item => self.renderItem(node, bounds),
        .modal => self.renderModal(node, bounds),

        .icon => {},
    }
}

This pattern is simple and extensible. Adding a new node kind means:

  1. Add it to the Kind enum
  2. Add a case to this switch
  3. Implement the render function

Layout: The Column Algorithm
#

The column layout is the most important - it’s how we stack elements vertically with flexbox-like behavior.

The Algorithm
# 


              Available Space            
                                         
     
      Fixed Child (e.g., H1)           <- Measured height
     
  ~~~~~~~~~ gap ~~~~~~~~~                
     
      Fixed Child (e.g., text)         <- Measured height
     
  ~~~~~~~~~ gap ~~~~~~~~~                
     
                                      
      Flex Child (flex=1)              <- Gets remaining space

Steps:

  1. Measure all fixed-size children
  2. Calculate remaining space after gaps
  3. Distribute remaining space to flex children
  4. Position and render each child

fn renderCol(self: *Renderer, node: Node, bounds: Rect) void {
    // 
    // Step 1: Apply padding to get content area
    // 
    const content = bounds.inset(node.props.pad);
    const gap = node.props.gap;
    const n = node.children.len;

    // Empty container - nothing to do
    if (n == 0) return;

    // 
    // Step 2: Measure children
    // 
    //
    // We need to know:
    // - Total flex units (sum of all flex values)
    // - Total fixed height (sum of non-flex children heights)
    //
    var flex_total: i32 = 0;
    var fixed_height: i32 = 0;

    for (node.children) |child| {
        if (child.props.flex > 0) {
            // This child wants to flex
            flex_total += child.props.flex;
        } else {
            // This child has a fixed/measured size
            fixed_height += self.measureH(child);
        }
    }

    // 
    // Step 3: Calculate flex unit size
    // 
    //
    // Total gaps = gap * (n - 1)
    // Available for flex = content.height - gaps - fixed_height
    // Each flex unit = available / total_flex_units
    //
    const total_gaps = gap * @as(i32, @intCast(n - 1));
    const flex_space = @max(0, content.h - total_gaps - fixed_height);
    const flex_unit = if (flex_total > 0)
        @divFloor(flex_space, flex_total)
    else
        0;

    // 
    // Step 4: Position and render each child
    // 
    var y = content.y;

    for (node.children) |child| {
        // Determine this child's height
        const child_h = if (child.props.flex > 0)
            flex_unit * child.props.flex  // Flex: proportional share
        else
            self.measureH(child);         // Fixed: measured height

        // Create bounds for this child
        const child_bounds = Rect.init(
            content.x,      // Full width of content area
            y,              // Current Y position
            content.w,      // Full width
            child_h,        // Calculated height
        );

        // Recursively render the child
        self.renderNode(child, child_bounds);

        // Advance Y position for next child
        y += child_h + gap;
    }
}

fn renderRow(self: *Renderer, node: Node, bounds: Rect) void {
    const content = bounds.inset(node.props.pad);
    const gap = node.props.gap;
    const n = node.children.len;

    if (n == 0) return;

    // Measure WIDTH instead of height
    var flex_total: i32 = 0;
    var fixed_width: i32 = 0;

    for (node.children) |child| {
        if (child.props.flex > 0) {
            flex_total += child.props.flex;
        } else {
            fixed_width += self.measureW(child);  // Width!
        }
    }

    const total_gaps = gap * @as(i32, @intCast(n - 1));
    const flex_space = @max(0, content.w - total_gaps - fixed_width);
    const flex_unit = if (flex_total > 0)
        @divFloor(flex_space, flex_total)
    else
        0;

    // Position horizontally
    var x = content.x;

    for (node.children) |child| {
        const child_w = if (child.props.flex > 0)
            flex_unit * child.props.flex
        else
            self.measureW(child);

        const child_bounds = Rect.init(
            x,              // Current X position
            content.y,      // Same Y as content
            child_w,        // Calculated width
            content.h,      // Full height
        );

        self.renderNode(child, child_bounds);

        x += child_w + gap;
    }
}

/// Measure the natural height of a node
fn measureH(self: *Renderer, node: Node) i32 {
    _ = self;
    return switch (node.kind) {
        // Text: height depends on font size
        .text => switch (node.props.size) {
            .xs => 16,
            .sm => 18,
            .md => 22,
            .lg => 26,
            .xl => 32,
            .xxl => 40,
        },

        // Interactive elements have fixed heights
        .button, .toggle, .input => 36,

        // Divider is a thin line with padding
        .divider => 16,

        // Spacer has no intrinsic size (it's flex)
        .spacer => 0,

        // Other elements get a default
        else => 32,
    };
}

/// Measure the natural width of a node
fn measureW(self: *Renderer, node: Node) i32 {
    _ = self;
    return switch (node.kind) {
        // Buttons have a minimum width
        .button => 100,
        .toggle => 100,

        // Spacer is flexible
        .spacer => 0,

        // Default width
        else => 100,
    };
}

Note: A production system would measure text width based on the actual string and font. For simplicity, we use fixed estimates.

fn renderText(self: *Renderer, node: Node, bounds: Rect) void {
    // Get label text
    const label = node.props.label orelse return;

    // Convert to null-terminated string (raylib requirement)
    const slice = self.toZ(label);

    // Map size enum to pixel size
    const font_size: i32 = switch (node.props.size) {
        .xs => 11,
        .sm => 13,
        .md => 15,
        .lg => 18,
        .xl => 24,
        .xxl => 32,
    };

    // Resolve color (default to primary text color)
    const color = self.resolveColor(node.props.color);

    // Draw!
    rl.drawText(slice, bounds.x, bounds.y, font_size, color);
}

/// Convert slice to null-terminated string
fn toZ(self: *Renderer, str: []const u8) [:0]const u8 {
    const len = @min(str.len, self.text_buf.len - 1);
    @memcpy(self.text_buf[0..len], str[0..len]);
    self.text_buf[len] = 0;
    return self.text_buf[0..len :0];
}

/// Convert semantic color to actual RGB
fn resolveColor(self: *Renderer, c: ?Node.Color) rl.Color {
    _ = self;
    return if (c) |color| switch (color) {
        .primary => Theme.text,
        .secondary => Theme.text_secondary,
        .muted => Theme.text_muted,
        .accent => Theme.accent,
        .success => Theme.success,
        .warning => Theme.warning,
        .danger => Theme.danger,
        .bg => Theme.bg,
        .bg_secondary => Theme.bg_secondary,
        .border => Theme.border,
        .white => rl.Color.white,
        .black => rl.Color.black,
    } else Theme.text;
}

        Theme.divider,   // Color

}

fn renderButton(self: *Renderer, node: Node, bounds: Rect) void {
    // 
    // Step 1: Determine interaction state
    // 
    const hover = self.isHovered(bounds);
    const press = hover and rl.isMouseButtonDown(.left);

    // 
    // Step 2: Choose background color based on state
    // 
    //
    // Priority: disabled > pressed > hovered > normal
    //
    const bg = if (node.props.disabled)
        Theme.bg_disabled       // Grayed out
    else if (press)
        Theme.button_active     // Darker when pressed
    else if (hover)
        Theme.button_hover      // Lighter when hovered
    else
        Theme.button_bg;        // Normal state

    // 
    // Step 3: Draw the button background
    // 
    rl.drawRectangleRounded(
        bounds.toRaylib(),
        0.2,    // Corner roundness (0.0 - 1.0)
        8,      // Number of segments for rounded corners
        bg,     // Color
    );

    // 
    // Step 4: Draw the label centered
    // 
    if (node.props.label) |label| {
        const slice = self.toZ(label);

        // Measure text to center it
        const text_width = rl.measureText(slice, 15);
        const tx = bounds.x + @divFloor(bounds.w - text_width, 2);
        const ty = bounds.y + @divFloor(bounds.h - 15, 2);

        // Text color depends on disabled state
        const text_color = if (node.props.disabled)
            Theme.text_disabled
        else
            Theme.text;

        rl.drawText(slice, tx, ty, 15, text_color);
    }

    // 
    // Step 5: Handle click
    // 
    //
    // Only register click if:
    // - Not disabled
    // - Mouse is over the button
    // - Mouse button was JUST pressed (not held)
    //
    if (!node.props.disabled and hover and rl.isMouseButtonPressed(.left)) {
        if (node.props.on_press) |handler_idx| {
            self.queueEvent(handler_idx);
        }
    }
}

/// Check if mouse is over a rectangle
fn isHovered(self: *Renderer, bounds: Rect) bool {
    _ = self;
    const pos = rl.getMousePosition();
    const mx: i32 = @intFromFloat(pos.x);
    const my: i32 = @intFromFloat(pos.y);

    return mx >= bounds.x and
           mx < bounds.right() and
           my >= bounds.y and
           my < bounds.bottom();
}

/// Add event to queue (called later by flush())
fn queueEvent(self: *Renderer, idx: usize) void {
    if (self.event_count < self.events.len) {
        self.events[self.event_count] = idx;
        self.event_count += 1;
    }
}

fn renderToggle(self: *Renderer, node: Node, bounds: Rect) void {
    // Toggle has a fixed width
    const toggle_bounds = Rect.init(bounds.x, bounds.y, 100, bounds.h);
    const hover = self.isHovered(toggle_bounds);

    // Background color based on active state
    const bg = if (node.props.active)
        Theme.button_active     // Darker when active
    else if (hover)
        Theme.button_hover
    else
        Theme.button_bg;

    // Draw background
    rl.drawRectangleRounded(toggle_bounds.toRaylib(), 0.15, 8, bg);

    // Draw accent border when active
    if (node.props.active) {
        rl.drawRectangleRoundedLinesEx(
            toggle_bounds.toRaylib(),
            0.15,
            8,
            2,              // Border thickness
            Theme.accent,   // Accent color
        );
    }

    // Draw label
    if (node.props.label) |label| {
        const slice = self.toZ(label);
        const text_w = rl.measureText(slice, 15);
        const tx = toggle_bounds.x + @divFloor(toggle_bounds.w - text_w, 2);
        const ty = toggle_bounds.y + @divFloor(toggle_bounds.h - 15, 2);

        // Active toggles have brighter text
        const text_color = if (node.props.active)
            Theme.text
        else
            Theme.text_secondary;

        rl.drawText(slice, tx, ty, 15, text_color);
    }

    // Handle click -> queue on_change event
    if (hover and rl.isMouseButtonPressed(.left)) {
        if (node.props.on_change) |idx| {
            self.queueEvent(idx);
        }
    }
}

const Theme = struct {
    // 
    // Background Colors
    // 

    /// Primary background (main screen)
    const bg = rl.Color.init(30, 30, 30, 255);

    /// Secondary background (cards, elevated surfaces)
    const bg_secondary = rl.Color.init(45, 45, 45, 255);

    /// Hover state background
    const bg_hover = rl.Color.init(55, 55, 55, 255);

    /// Selected item background

Execution:

1. render() called
   - Clear event queue
   - Clear screen with Theme.bg

2. renderNode(col, screen_bounds)
   - Detect kind = .col
   - Call renderCol()

3. renderCol() for root column
   - bounds = (0, 0, 800, 600)
   - Apply padding: content = (20, 20, 760, 560)
   - gap = 16
   - Measure children:
     - h1: 40px
     - text: 32px
     - row: 36px
     - button: 36px
     - Total fixed: 144px
   - No flex children, flex_space = 0
   - Render children:

4.   renderNode(h1, (20, 20, 760, 40))
     - kind = .text
     - Call renderText()
     - Draw "Counter" at (20, 20), size 32, color white

5.   renderNode(text "42", (20, 76, 760, 32))
     - kind = .text
     - Draw "42" at (20, 76), size 24, color white

6.   renderNode(row, (20, 124, 760, 36))
     - kind = .row
     - Call renderRow()

7.     renderRow() for button row
       - content = (20, 124, 760, 36)
       - gap = 8
       - Measure children:
         - button "-": 100px
         - button "+": 100px
         - Total fixed: 200px
       - Render children:

8.       renderNode(button "-", (20, 124, 100, 36))
         - kind = .button
         - Call renderButton()
         - Check hover: mouse at (50, 135)? Yes!
         - bg = button_hover
         - Draw rounded rect
         - Draw "-" centered
         - Mouse not pressed, no event queued

9.       renderNode(button "+", (128, 124, 100, 36))
         - kind = .button
         - Not hovered
         - bg = button_bg
         - Draw rounded rect
         - Draw "+" centered

10.  renderNode(button "Reset", (20, 176, 760, 36))
     - kind = .button
     - Not hovered
     - Draw rounded rect
     - Draw "Reset" centered

11. render() complete
    - Frame displayed
